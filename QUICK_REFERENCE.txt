-- Get properties with owner details
SELECT p.title, p.price, p.city, u.first_name, u.last_name
FROM properties p
LEFT JOIN users u ON p.owner_id = u.id
WHERE p.available = true;

-- Get property with all images
SELECT p.title, i.image_url, i.caption, i.is_primary
FROM properties p
LEFT JOIN property_images i ON p.id = i.property_id
WHERE p.id = 1
ORDER BY i.display_order;

-- Count properties per user
SELECT u.first_name, u.last_name, COUNT(p.id) as property_count
FROM users u
LEFT JOIN properties p ON u.id = p.owner_id
GROUP BY u.id, u.first_name, u.last_name;
